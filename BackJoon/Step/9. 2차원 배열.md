```java
import java.util.Scanner;

public class Main {

    public static void main(String[] args) throws Exception{

        Scanner sc = new Scanner(System.in);

        int A = sc.nextInt();
        int B = sc.nextInt();

        int[][] array = new int[A][B];
        int[][] array2 = new int[A][B];

        for (int i = 0; i < A; i++){
            for (int j = 0; j < B; j++){
                array[i][j] = sc.nextInt();
            }
        }

        for (int i = 0; i < A; i++){
            for (int j = 0; j < B; j++){
                array2[i][j] = sc.nextInt();
            }
        }

        for (int i = 0; i < A; i++){
            for (int j = 0; j < B; j++){
                System.out.print(array[i][j] + array2[i][j] + " ");
                if (j == B-1){
                    System.out.println();
                }
            }
        }
    }
}
```

```java
import java.util.Scanner;

public class Main {

    public static void main(String[] args) throws Exception{

        Scanner sc = new Scanner(System.in);

        int[][] array = new int[9][9];

        int max = 0;
        int x = 0, y = 0;

        for (int i = 0; i < 9; i++){
            for (int j = 0; j < 9; j++){
                array[i][j] = sc.nextInt();

                if (max < array[i][j])
                {
                    max = array[i][j];
                    x = i; y = j;
                }

            }
        }
        System.out.println(max);
        System.out.println((x + 1) + " " + (y + 1));

    }
}
```

```java

import java.util.Scanner;

public class Main {

    public static void main(String[] args) throws Exception{

        Scanner sc = new Scanner(System.in);

        int[][] array = new int[100][100];

        int N = sc.nextInt();

        int count = 0;

        for (int i = 0; i < N; i++){

            int A = sc.nextInt();
            int B = sc.nextInt();

            for (int j = A; j < A + 10; j++){
                for (int k = B; k < B + 10; k++){
                    array[j][k] = 1;
                }
            }
        }

        for (int i = 0; i < 100; i++){
            for (int j = 0; j < 100; j++){
                if(array[i][j] == 1){
                    count++;
                }
            }
        }
        System.out.println(count);

    }

}
```
