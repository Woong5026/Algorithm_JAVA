        int max = 0;
        int cnt = 0;

        for (int i = 0; i < n; i++){
            if (arr[i] > max){
                max = arr[i];
                cnt++;

                if (cnt == m){
                    break;
                }else if(cnt < m){
                    max = 0;
                }

            }
        }
        System.out.println(max);

    }
}
